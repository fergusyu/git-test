select ASTTC.TC001 AS '單別',ASTTC.TC002 AS '單號',ASTTC.TC004 AS '資產編號',ASTMB.MB002  AS '資產名稱',ASTMB.MB003 AS '資產規格',ASTTC.TC008 AS '折舊費用',ASTTD.TD003 AS '部門代號',CMSME.ME002 AS '部門名稱',ASTTD.TD006 AS '折舊科目',ACTMA.MA003 AS '科目名稱',ASTMB.MB006 AS '資產類別',ASTTC.TC027 AS '單據日期',ASTTC.TC023 AS '傳票單別',ASTTC.TC024 AS '傳票單號',ASTTC.TC031 AS '產生分路',ASTTC.TC015 AS '確認碼',ASTTC.TC003 AS '異動日期',ASTTC.TC028 AS '異動者',ASTTC.TC025 AS '自動銷帳'
from ASTTC JOIN ASTTD ON ASTTC.TC002=ASTTD.TD002 JOIN ACTMA ON ASTTD.TD006=ACTMA.MA001 JOIN ASTMB ON ASTTC.TC004=ASTMB.MB001 JOIN CMSME ON ASTTD.TD003=CMSME.ME001
WHERE ASTTC.TC001 IN ('C601','C602')
